```C++
/* 在C++中，輸出指定格式內容 */
printf("字元輸出：%c , %c \n","a", chara);
printf("整數(int)輸出：%d , %d \n", 2, num);
printf("浮點數(float)輸出：%f , %f \n", 2.125, flt);
//要注意使用浮點數(float)格式輸出會預設6個小數點，因此上面的輸出結果為 2.125000

/* 若格式如右：%3d，代表輸出整數，會有3個欄位*/
printf("整數(int)輸出：%3d , %d \n", 2, num);
//輸出結果為->□□2 , num的值(□代表空格)

/* 
若格式如右：%3.2f，代表輸出浮點數，此數值有3個欄位，小數點只會出現後兩位並四捨五入到小數點第二位*/
printf("整數(int)輸出：%3.2f , %f \n", 2.125, flt);
//輸出結果為->2.13 , flt的值(□代表空格)

/*若欄位設定的值比輸入值的長度還要小，則會以沒有來看待*/
printf("整數(int)輸出：%1.2f , %f \n", 2.125, flt);
/*
由上可知，第一個數值欄位設定成1個，但是輸入的值長度為5(包含小數點)，因此程式會把%1.2f當成%.2f看待
*/
//所以輸出結果為->2.13 , flt
```
